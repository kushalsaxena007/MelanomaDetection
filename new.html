import torch
import torchvision.transforms as transforms
from torch.utils.data import Dataset
from torchvision.datasets import ImageFolder
from PIL import Image

class CustomDataset(Dataset):
    def __init__(self, folder_path):
        self.data = ImageFolder(folder_path)

    def __getitem__(self, index):
        x, y = self.data[index]
        return x, y

    def __len__(self):
        return len(self.data)

# Define the paths to your train and test folders
train_folder = '/kaggle/input/skin-cancer-malignant-vs-benign/train/'
test_folder = '/kaggle/input/skin-cancer-malignant-vs-benign/test/'

# Define transformations for the dataset
transform = transforms.Compose([
    transforms.Resize((64, 64)),  # Resize the images to a consistent size
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.5, 0.5, 0.5], std=[0.5, 0.5, 0.5])  # Normalize the image tensors
])

# Create train and test datasets
train_dataset = CustomDataset(train_folder)
test_dataset = CustomDataset(test_folder)

# Create separate lists for inputs (images) and labels
x_train = []
y_train = []
for data in train_dataset:
    image, label = data
    image = transform(image)  # Apply transform here
    x_train.append(image)
    y_train.append(label)

x_test = []
y_test = []
for data in test_dataset:
    image, label = data
    image = transform(image)  # Apply transform here
    x_test.append(image)
    y_test.append(label)

# Convert the lists to tensors
x_train = torch.stack(x_train)
y_train = torch.tensor(y_train)
x_test = torch.stack(x_test)
y_test = torch.tensor(y_test)

import numpy as np
from skimage.feature import local_binary_pattern
from skimage.color import rgb2gray

# Define the LBP parameters
radius = 3
n_points = 8 * radius

# Initialize lists to store the LBP features
lbp_train = []
lbp_test = []

# Extract LBP features from x_train
for image in x_train:
    # Convert the tensor to a NumPy array and then to grayscale
    image_np = image.permute(1, 2, 0).numpy()
    image_gray = rgb2gray(image_np)

    # Compute LBP features
    lbp = local_binary_pattern(image_gray, n_points, radius, method='uniform')

    # Flatten the LBP features and append to the list
    lbp_train.append(lbp.flatten())

# Extract LBP features from x_test
for image in x_test:
    # Convert the tensor to a NumPy array and then to grayscale
    image_np = image.permute(1, 2, 0).numpy()
    image_gray = rgb2gray(image_np)

    # Compute LBP features
    lbp = local_binary_pattern(image_gray, n_points, radius, method='uniform')

    # Flatten the LBP features and append to the list
    lbp_test.append(lbp.flatten())

# Convert the LBP feature lists to NumPy arrays
lbp_train = np.array(lbp_train)
lbp_test = np.array(lbp_test)
